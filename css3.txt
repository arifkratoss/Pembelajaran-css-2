Cara menggunaka border radius {
    border-radius=untuk semua sisi
    border-top-right/left=untuk sisi yang atas kanan atau kiri
    border-bottom-right/left=untuk sisi yang bawah bagian kanan/kiri
    border-radius=20% 0 20% 0=untuk meggunaka keseluruhan tanpa harus mengetik codenya lagi
}

.box-shadow{
    -properti yang harus di dugunakan<x-offset><y-offset><blur><color>
    -x-offset=adalah mengatur secara horizontal,apabila di kasih minus akan bergerak ke kiri
    -y-offset=adalah jarak bayangan secara vertical
    -blur=seberapa banyak blur yang ingin kita kasih ke bayanagannnya
    -color=adalah warna yang ingin kita kasih
    -spread=ukuran dari bayannganya
    -inset=bayangan di dalam kotak
    -multiple-shadow=kita bisa memebrikan lebih dari satu bayangan kepada satu elemen
}
